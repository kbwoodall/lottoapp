function displayTime() {
  var str = "";
  var str2 = "";
  var ampm = "";

  var currentTime = new Date();
  var hours = currentTime.getHours();
  var minutes = currentTime.getMinutes();
  var seconds = currentTime.getSeconds();

  if (minutes < 10) {
    minutes = "0" + minutes;
  }
  if (seconds < 10) {
    seconds = "0" + seconds;
  }
  str += hours + ":" + minutes + ":" + seconds + " ";
  if (hours > 11) {
    str += "PM";
    ampm = "PM";
  } else {
    str += "AM";
    ampm = "PM";
  }
  if (hours > 12) {
    hours = hours - 12;
    str2 += hours + ":" + minutes + ":" + seconds + " ";
  }
  return (str2 += " " + ampm);
}

function doLotto() {
  const saveNumbers = [];

  const arrayFrom1To49 = Array.from({ length: 49 }, (_, index) => index + 1);

  const randomIndex = Math.floor(Math.random() * arrayFrom1To49.length);
  const randomNumber = arrayFrom1To49[randomIndex];
  saveNumbers.push(randomNumber);

  const numberToRemove2 = randomNumber;
  const newList2 = arrayFrom1To49.filter(
    (number) => number !== numberToRemove2
  );

  const randomIndex2 = Math.floor(Math.random() * newList2.length);
  const randomNumber2 = arrayFrom1To49[randomIndex2];
  saveNumbers.push(randomNumber2);

  const numberToRemove3 = randomNumber2;
  const newList3 = newList2.filter((number) => number !== numberToRemove3);

  const randomIndex3 = Math.floor(Math.random() * newList3.length);
  const randomNumber3 = newList3[randomIndex3];
  saveNumbers.push(randomNumber3);

  const numberToRemove4 = randomNumber3;
  const newList4 = newList3.filter((number) => number !== numberToRemove4);

  const randomIndex4 = Math.floor(Math.random() * newList4.length);
  const randomNumber4 = newList4[randomIndex4];
  saveNumbers.push(randomNumber4);

  const numberToRemove5 = randomNumber4;
  const newList5 = newList4.filter((number) => number !== numberToRemove5);

  const randomIndex5 = Math.floor(Math.random() * newList5.length);
  const randomNumber5 = newList5[randomIndex5];
  saveNumbers.push(randomNumber5);

  const numberToRemove6 = randomNumber5;
  const newList6 = newList5.filter((number) => number !== numberToRemove6);

  const randomIndex6 = Math.floor(Math.random() * newList6.length);
  const randomNumber6 = newList6[randomIndex6];
  saveNumbers.push(randomNumber6);

  return (
    <div className="App">
      <header className="App-header">
        <h2>Select 6 random numbers for Lotto</h2>
        <h3>Refresh screen for new numbers</h3>

        <div>{arrayFrom1To49}</div>
        <div>{newList2}</div>
        <div>{newList3}</div>
        <div>{newList4}</div>
        <div>{newList5}</div>
        <div>{newList6}</div>
        <p>
          {saveNumbers[0]} &nbsp;
          {saveNumbers[1]} &nbsp;
          {saveNumbers[2]} &nbsp;
          {saveNumbers[3]} &nbsp;
          {saveNumbers[4]} &nbsp;
          {saveNumbers[5]} &nbsp;
        </p>

        <p className="text-yellow-400"> {displayTime()}</p>
      </header>
    </div>
  );
}
